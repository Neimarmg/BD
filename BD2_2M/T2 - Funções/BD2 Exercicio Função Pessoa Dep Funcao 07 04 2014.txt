CREATE OR REPLACE FUNCTION get_totalSalario(integer)
	RETURNS decimal AS $$

	DECLARE totalSalario decimal(10,2);

	BEGIN
		SELECT into totalSalario SUM(pessoa_fisica.salario)
		
		FROM pessoa_fisica
		WHERE pessoa_fisica.id_pessoa = $1;
		
		RETURN totalSalario;
	END;
$$ LANGUAGE 'plpgsql';
