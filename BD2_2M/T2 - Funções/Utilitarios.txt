		
CREATE TABLE utilitarios (
	idUtilitarios serial primary key,	
	nomeUtilitario varChar (100) null,
	idTipoUtilitario integer null,
	CHECK(
		idTipoUtilitario > 0
		),
	CONSTRAINTS FOREIGN key (idUtilitarios) REFERENCES tipoUtilitarios(idTipoUtilitario)
)
