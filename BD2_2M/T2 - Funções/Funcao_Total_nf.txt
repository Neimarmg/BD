
CREATE FUNCTION get_tot_NF() RETURNS SETOF RECORD AS $$
BEGIN
RETURN QUERY SELECT SUM(intensNF.qtd * material.custo) AS Custo
	FROM material, idItensNf 
    WHERE idItensNf.idNf = notaFiscal.idNf;
RETURN;
END;
$$ LANGUAGE 'plpgsql';

--Chamada da Função
SELECT * FROM get_tot_NF() AS (
	Custo DECIMAL(10,2);
)
