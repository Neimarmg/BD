CREATE TABLE material (
	idMaterial SERIAL PRIMARY KEY,
	nome VARCHAR(200),
	qtd DECIMAL (10,2),
	custo DECIMAL (10,2),
	idMarca INTEGER,
	idTipo INTEGER,
	CONSTANT fk_marca FOREIGN KEY (idMarca) REFERENCES utilitarios (idUtilitarios)
	CONSTANT fk_tipo FOREIGN KEY (idTipo) REFERENCES utilitarios (idUtilitarios)
);
