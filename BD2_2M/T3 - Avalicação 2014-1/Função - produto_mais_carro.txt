CREATE or replace FUNCTION get_produto_mais_carro() RETURNS TABLE (
nomeProduto VARCHAR (15),
preco DOUBLE (10,2)) AS $$
BEGIN
RETURN 
	QUERY SELECT INTO
		produtos.nome, 
		produtos.preco_unitario
		FROM produtos 
        WHERE MAX(produtos.preco_unitario);
RETURN;
END;
$$ LANGUAGE 'plpgsql'


SELECT * FROM get_produto_mais_carro();
